<script>
        // Simple micro-interaction for the star toggle and other UI elements
        document.addEventListener('DOMContentLoaded', () => {
            const quantityInput = document.querySelector('input[type="number"]');
            const plusBtn = quantityInput.nextElementSibling;
            const minusBtn = quantityInput.previousElementSibling;

            plusBtn.onclick = () => quantityInput.value = parseInt(quantityInput.value) + 1;
            minusBtn.onclick = () => {
                if(parseInt(quantityInput.value) > 1) quantityInput.value = parseInt(quantityInput.value) - 1;
            };

            // Mimic close behavior
            const closeBtn = document.querySelector('button .material-symbols-outlined[data-icon="close"]').parentElement;
            closeBtn.onclick = () => {
                document.querySelector('.animate-in').classList.add('translate-x-full');
                document.querySelector('.animate-in').classList.remove('slide-in-from-right');
                setTimeout(() => window.history.back(), 300);
            };
        });
