<script>
    // Star toggle interaction
    document.querySelectorAll('.inventory-row button:first-of-type').forEach(btn => {
        btn.addEventListener('click', function(e) {
            e.stopPropagation();
            const icon = this.querySelector('span');
            const isFilled = icon.classList.contains('filled-icon');
            
            if (isFilled) {
                icon.classList.remove('filled-icon');
                this.classList.replace('text-primary', 'text-primary/40');
            } else {
                icon.classList.add('filled-icon');
                this.classList.replace('text-primary/40', 'text-primary');
            }
        });
    });

    // Add to list interaction
    document.querySelectorAll('.inventory-row button:last-of-type').forEach(btn => {
        btn.addEventListener('click', function(e) {
            e.stopPropagation();
            const toast = document.getElementById('toast');
            toast.classList.remove('opacity-0', 'translate-y-4');
            toast.classList.add('opacity-100', 'translate-y-0');
            
            setTimeout(() => {
                toast.classList.add('opacity-0', 'translate-y-4');
                toast.classList.remove('opacity-100', 'translate-y-0');
            }, 2000);
        });
    });
</script>
