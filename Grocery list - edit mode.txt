<script>
    // Handle checkbox change
    document.querySelectorAll('.check-pill').forEach(checkbox => {
        checkbox.addEventListener('change', function() {
            const row = this.closest('.grocery-row');
            const textContainer = row.querySelector('.flex-grow');
            if (this.checked) {
                textContainer.classList.add('item-checked');
                row.style.opacity = '0.6';
                row.classList.add('bg-surface-container-low');
            } else {
                textContainer.classList.remove('item-checked');
                row.style.opacity = '1';
                row.classList.remove('bg-surface-container-low');
            }
        });
    });

    // Handle Edit Mode Toggle (Long Press)
    let pressTimer;
    const body = document.body;
    const overlay = document.getElementById('overlayExit');

    const enterEditMode = () => {
        body.classList.add('edit-mode');
        if (window.navigator.vibrate) window.navigator.vibrate(40);
    };

    const exitEditMode = () => {
        body.classList.remove('edit-mode');
    };

    document.querySelectorAll('.grocery-row').forEach(row => {
        const startPress = (e) => {
            if (body.classList.contains('edit-mode')) return;
            // Don't trigger if clicking child buttons
            if (e.target.closest('button') || e.target.closest('label')) return;
            
            pressTimer = window.setTimeout(enterEditMode, 500);
        };

        const cancelPress = () => {
            clearTimeout(pressTimer);
        };

        row.addEventListener('mousedown', startPress);
        row.addEventListener('touchstart', startPress, {passive: true});
        row.addEventListener('mouseup', cancelPress);
        row.addEventListener('mouseleave', cancelPress);
        row.addEventListener('touchend', cancelPress);
        row.addEventListener('touchmove', cancelPress);
    });

    // Exit mode by clicking overlay
    overlay.addEventListener('click', exitEditMode);

    // Prevent propagation on list area to keep it active
    document.querySelector('main').addEventListener('click', (e) => {
        if (body.classList.contains('edit-mode')) {
            e.stopPropagation();
        }
    });

    // Quantity buttons mockup
    document.querySelectorAll('.stepper-btn').forEach(btn => {
        btn.addEventListener('click', (e) => {
            e.stopPropagation();
            const span = btn.parentElement.querySelector('span');
            let val = parseInt(span.textContent);
            if (btn.querySelector('.material-symbols-outlined').textContent === 'add') {
                span.textContent = val + 1;
            } else if (val > 1) {
                span.textContent = val - 1;
            }
        });
    });

    // Delete buttons mockup
    document.querySelectorAll('.edit-controls button.text-error').forEach(btn => {
        btn.addEventListener('click', (e) => {
            e.stopPropagation();
            const row = btn.closest('.grocery-row');
            row.style.transform = 'translateX(20px)';
            row.style.opacity = '0';
            setTimeout(() => row.remove(), 200);
        });
    });
</script>
