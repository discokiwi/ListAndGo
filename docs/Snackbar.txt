        function showAddedSnackbar() {
            const el = document.getElementById('added-snackbar');
            el.classList.remove('hidden');
            setTimeout(() => {
                el.classList.add('hidden');
            }, 4000);
        }

        function showDeleteSnackbar() {
            const el = document.getElementById('deleted-snackbar');
            el.classList.remove('hidden');
            // Longer duration for items with actions
            setTimeout(() => {
                el.classList.add('hidden');
            }, 6000);
        }

        function undoDelete() {
            // Logic for undoing delete would go here
            document.getElementById('deleted-snackbar').classList.add('hidden');
            showAddedSnackbar(); // Show visual feedback that it was restored
        }

            setTimeout(showAddedSnackbar, 1000);
